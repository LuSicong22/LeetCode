```go
func preorderTraversal(root *TreeNode) []int {
    res:= []int{}
    preorder(root, &res)
    return res
}

func preorder(root *TreeNode, output *[]int) {
    if root == nil {
        return
    }
    *output = append(*output, root.Val)
    preorder(root.Left, output)
    preorder(root.Right, output)
}
```
