class Solution {
    public String reverseStr(String s, int k) {
        int n= s.length();
        char[] arr = s.toCharArray();
        for(int i=0;i<n;i+=2*k) {
            reverse(arr,i,Math.min(i+k,n)-1);
        }
        return new String(arr);
    }

    public void reverse(char[] arr,int left,int right){
        while(left<right){
            char tmp=arr[left];
            arr[left]=arr[right];
            arr[right]=tmp;
            left++;
            right--;
        }
    }
}
