## Method 1: Helper stack

```java
class MinStack {
     private Deque<Integer> s;
     private Deque<Integer> mins;

    public MinStack() {
        s=new ArrayDeque<>();
        mins=new ArrayDeque<>();
        mins.push(Integer.MAX_VALUE);
    }
    
    public void push(int val) {
        s.push(val);
        mins.push(Math.min(val,mins.peek()));
    }
    
    public void pop() {
        s.pop();
        mins.pop();
    }
    
    public int top() {
        return s.peek();
    }
    
    public int getMin() {
        return mins.peek();
    }
}
```

## Method 2: One stack
```java
public class MinStack {
    private int min_val = Integer.MAX_VALUE;
    private Stack<Integer> s = new Stack<>();
    
    public MinStack() {}  
    public void push(int x) {
        if (x <= min_val) {
            s.push(min_val);
            min_val = x;
        }
        s.push(x);
    }    
    public void pop() {
        if (s.pop() == min_val) min_val = s.pop();
    }   
    public int top() {
        return s.peek();
    }    
    public int getMin() {
        return min_val;
    }
}
```
