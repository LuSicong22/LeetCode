### Merge Sort
Java
```java
//归并排序
class Solution {
    public int[] sortArray(int[] nums) {
        //归并排序
        mergeSort(nums,0,nums.length-1);
        //返回结果
        return nums;
    }

    public void mergeSort(int[] nums, int l, int r) {
        //终止条件
        if (l>=r){
            return;
        }
        //定义中点
        int mid=l+((r-l)>>1);
        //递归归并排序左半区
        mergeSort(nums,l,mid);
        //递归归并排序右半区
        mergeSort(nums,mid+1,r);
        //合并数组
        merge(nums,l,mid,r);
        
    }

    //合并数组函数
    public void merge(int[] nums, int l,int mid, int r){
        int tmp[]=new int[r-l+1];
        int p1=l;
        int p2=mid+1;
        int i=0;
        while(p1<=mid&&p2<=r){
            if (nums[p1]<=nums[p2]){
                tmp[i]=nums[p1];
                p1++;
            }else{
                tmp[i]=nums[p2];
                p2++;
            }
            i++;
        }
        while (p1<=mid){
            tmp[i]=nums[p1];
            p1++;
            i++;
        }
        while (p2<=r) {
            tmp[i]=nums[p2];
            p2++;
            i++;
        }
        for (int k=0;k<tmp.length;k++){
            nums[l+k]=tmp[k];
        }
    }
}
```
